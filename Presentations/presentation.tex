\begin{frame}[fragile]
	\frametitle{The Equation}
	
	\begin{equation*}
		\bar{L} = - P \alpha
	\end{equation*}
	
	\begin{itemize}
		\item \alert<1>{$\bar{L}$ Vectorized log light field}
		\item \alert<2>{$\alpha$ Vector holding unkowns}
	\end{itemize}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Optimization Problem}
	
	\begin{equation*}
		\begin{aligned}
			& \underset{\alpha}{\text{argmin}} 	& & \left\lVert P \alpha + \bar{L} \right\rVert ^2 \\
			& \text{subject to} 				& & \alpha \geq 0.
		\end{aligned}
	\end{equation*}
	
	\begin{itemize}
		\item \alert<1>{System is overdetermined}
		\item \alert<2>{Need iterative solver}
	\end{itemize}
\end{frame}
